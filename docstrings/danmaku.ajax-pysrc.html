 22      ''' 
 23      A toolkit method. 
 24      Divide a long list into separate pages, 
 25      very useful in multi-pages & carousel. 
 26   
 27      @param origin_list: The list to be split 
 28      @param page_len: How many items in a page 
 29   
 30      @return: A list of pages 
 31      ''' 
 32   
 33      page_list = [] 
 34      i = 0 
 35      while i + page_len <= len(origin_list): 
 36          page_list.append(origin_list[i:i + page_len]) 
 37          i += page_len 
 38
